	ACCOUNTID INT,
	NAME TEXT,
	ISCLIENT BOOLEAN
);

DROP TABLE IF EXISTS PROPERTIES;

	ID INT,
	ADDRESS TEXT,
	ORIGINALOWNER INT
);

	ID INT,
	BUYERID INT,
	SELLERID INT,
	PROPERTYID INT,
	AMOUNT FLOAT,
	FUNDSRECEIVED BOOLEAN,
	FUNDSRELEASED BOOLEAN
);
